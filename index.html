圆轨迹移动

var t = 0.05;
setInterval(function() {
    t += 0.05;
    object
        .disableTransition()
        .moveX(3 * Math.sin(t))
        .moveY(3 * Math.cos(t))
        .go();
}, 15);
Play Reset
