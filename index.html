Gate.js 是一个轻量级的 JavaScript 库，用来创建和管理 3D Transform，压缩后只有 5kb，它不依赖于任何第三方库。

我们可以直接通过 Gate.js 来创建 HTML 并指定它该执行什么动画。

旋转

x 旋转

object
    .enableTransition()
    .setRotateX(360)
    .go();

y 旋转

object
    .enableTransition()
    .setRotateY(360)
    .go();

z 旋转

object
    .enableTransition()
    .setRotateZ(360)
    .go();
Play Reset

x, y, z 旋转

object
    .enableTransition()
    .setRotate(180, 270, 360)
    .go();
Play Reset

圆轨迹移动

var t = 0.05;
    setInterval(function() {
    t += 0.05;
    object
    .disableTransition()
    .moveX(3 * Math.sin(t))
    .moveY(3 * Math.cos(t))
    .go();
    }, 15);
